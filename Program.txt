Program 
    Vars
        begin
        A = 0 ;
        B = 1 ; 
        end

    begin
        Handler1
            Regs
                begin
                X = 0 ; 
                Y = 0 ;
                end

            begin 

                Message1

                    begin
                    0 : Write B 2 ;
                    1 : Read X B ;
                    2 : Read Y B ;
                    end

                Message2

                    begin 
                    3 : Read X A ;
                    4 : Read Y A ;
                    5 : Write A 2 ;
                    end

            end
        Handler2
            Regs
                begin
                U = 0 ;
                Z = 0 ;
                end

            begin

            Message3
                begin
                6 : Read U A ;
                7 : Write A 1 ;
                8 : Read Z A ;
                end

            Message4
                begin
                9 : Post Handler1 Message2 ;
                10 : Write B 1 ;
                11 : Read Z B ;
                end

            end
    end
